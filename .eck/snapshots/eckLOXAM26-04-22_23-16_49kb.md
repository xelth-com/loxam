pub fn corrupt(data: &[u8]) -> Vec<u8> {
    let mut result = Vec::with_capacity(data.len() + data.len() / 128);
    for i in 0..data.len() {
        if data[i] == 0x0A && (i == 0 || data[i - 1] != 0x0D) {
            result.push(0x0D);
        }
        result.push(data[i]);
    }
    result
}

pub fn find_standalone_lf(data: &[u8]) -> Vec<usize> {
    let mut positions = Vec::new();
    for i in 0..data.len() {
        if data[i] == 0x0A && (i == 0 || data[i - 1] != 0x0D) {
            positions.push(i);
        }
    }
    positions
}

fn random_text(rng: &mut impl Rng, size: usize) -> Vec<u8> {
    let chars = b"abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789 .,!?;:'\"()-\n\r\t";
    let mut result = Vec::with_capacity(size);
    for _ in 0..size {
        result.push(chars[rng.gen_range(0..chars.len())]);
    }
    result
}

fn run_stress(runs: usize, size: usize) -> Result<()> {
    println!("=== Stress Test: {} runs, file size {} ===\n", runs, size);
    let mut rng = rand::thread_rng();

    let mut ok = 0usize;
    let mut fail = 0usize;
    let mut perfect = 0usize;
    let mut natural_crlf_total = 0usize;
    let mut strategy_counts: std::collections::HashMap<String, usize> = std::collections::HashMap::new();

    for run in 0..runs {
        let file_a = random_text(&mut rng, size);
        let file_b = random_text(&mut rng, size / 2);
        let file_c = random_text(&mut rng, size * 2);

        let files: Vec<(&str, &[u8])> = vec![
            ("a.txt", &file_a),
            ("b.txt", &file_b),
            ("c.txt", &file_c),
        ];

        let original_zip = zip_util::create_zip(&files);
        let natural_crlf = count_natural_crlf(&original_zip);
        natural_crlf_total += natural_crlf;

        let corrupted = corrupt::corrupt(&original_zip);

        match recover::recover(&corrupted) {
            Ok(result) => {
                *strategy_counts.entry(result.strategy.clone()).or_insert(0) += 1;
                if result.data == original_zip {
                    ok += 1;
                    perfect += 1;
                } else if validate_partial(&result.data) {
                    ok += 1;
                } else {
                    fail += 1;
                    println!("Run {}: recovery produced invalid ZIP (natural CRLFs: {})", run + 1, natural_crlf);
                }
            }
            Err(e) => {
                fail += 1;
                println!("Run {}: FAILED - {} (natural CRLFs: {})", run + 1, e, natural_crlf);
            }
        }

        if (run + 1) % 10 == 0 {
            println!(
                "Progress: {}/{} (ok={}, fail={}, natural_crlf_avg={:.1})",
                run + 1, runs, ok, fail, natural_crlf_total as f64 / (run + 1) as f64
            );
        }
    }

    println!("\n=== Results ===");
    println!("Total: {} | Perfect: {} | OK: {} | Failed: {}", runs, perfect, ok, fail);
    println!("Average natural CRLFs: {:.2}", natural_crlf_total as f64 / runs as f64);
    println!("Strategies used: {:?}", strategy_counts);

    if fail > 0 {
        anyhow::bail!("{} out of {} runs failed", fail, runs);
    }
    Ok(())
}

fn count_natural_crlf(data: &[u8]) -> usize {
    let mut count = 0;
    for i in 0..data.len().saturating_sub(1) {
        if data[i] == 0x0D && data[i + 1] == 0x0A {
            count += 1;
        }
    }
    count
}

fn count_diffs(a: &[u8], b: &[u8]) -> usize {
    let min_len = a.len().min(b.len());
    let mut count = a.len().abs_diff(b.len());
    for i in 0..min_len {
        if a[i] != b[i] {
            count += 1;
        }
    }
    count
}

fn validate_partial(data: &[u8]) -> bool {
    zip_util::is_valid_zip_signature(data)
        && matches!(zip_util::parse_and_validate(data), Ok(p) if p.entries.iter().all(|e| e.crc32_actual.map_or(false, |a| a == e.crc32_expected)))
}

fn print_extracted_files(data: &[u8]) -> Result<()> {
    let parsed = zip_util::parse_and_validate(data)?;
    for entry in &parsed.entries {
        let crc_ok = entry
            .crc32_actual
            .map_or(false, |a| a == entry.crc32_expected);
        println!(
            "  {} - {} bytes, CRC32 {} ({:08X})",
            entry.name,
            entry.compressed_size,
            if crc_ok { "OK" } else { "FAIL" },
            entry.crc32_expected,
        );
    }
    Ok(())
}

fn run_corrupt(input: &str, output: &str) -> Result<()> {
    let data = std::fs::read(input).context("Failed to read input file")?;
    println!("Input: {} bytes", data.len());

    let corrupted = corrupt::corrupt(&data);
    println!("Corrupted: {} bytes (+{})", corrupted.len(), corrupted.len() - data.len());

    let standalone = corrupt::find_standalone_lf(&data);
    println!("Replaced {} standalone LF bytes", standalone.len());

    std::fs::write(output, &corrupted)?;
    println!("Written to: {}", output);
    Ok(())
}

fn run_recover(input: &str, output: &str) -> Result<()> {
    let data = std::fs::read(input).context("Failed to read input file")?;
    println!("Input: {} bytes", data.len());

    let result = recover::recover(&data).context("Recovery failed")?;

    println!("\nStrategy: {}", result.strategy);
    println!("Attempts: {}", result.attempts);
    println!("Recovered: {} bytes", result.data.len());

    std::fs::write(output, &result.data)?;
    println!("Written to: {}", output);

    let parsed = zip_util::parse_and_validate(&result.data)?;
    for entry in &parsed.entries {
        let crc_ok = entry
            .crc32_actual
            .map_or(false, |a| a == entry.crc32_expected);
        println!(
            "  {} - {} bytes compressed, CRC32 {} ({:08X})",
            entry.name,
            entry.compressed_size,
            if crc_ok { "OK" } else { "FAIL" },
            entry.crc32_expected,
        );
    }

    Ok(())
}

use crate::zip_util;
use anyhow::{bail, Result};
use std::time::Instant;

pub struct RecoveryResult {
    pub data: Vec<u8>,
    pub strategy: String,
    pub attempts: u64,
}

pub fn recover(data: &[u8]) -> Result<RecoveryResult> {
    let crlf_positions = zip_util::find_crlf_positions(data);

    if crlf_positions.is_empty() {
        if validate_as_zip(data) {
            return Ok(RecoveryResult {
                data: data.to_vec(),
                strategy: "no_corruption".to_string(),
                attempts: 0,
            });
        }
        bail!("No CRLF sequences found and data is not a valid ZIP");
    }

    eprintln!(
        "Found {} CRLF positions in {} bytes",
        crlf_positions.len(),
        data.len()
    );

    let t = Instant::now();
    let lookup = CrlfLookup::new(&crlf_positions, data.len());

    if let Some(result) = try_remove_all(data, &crlf_positions, &lookup) {
        eprintln!("  Done in {:.2}s", t.elapsed().as_secs_f64());
        return Ok(result);
    }

    eprintln!(
        "'Remove all' not valid ({:.2}s), trying per-file targeted fix...",
        t.elapsed().as_secs_f64()
    );

    if let Some(result) = try_per_file_fix(data, &crlf_positions, &lookup) {
        eprintln!("  Done in {:.2}s", t.elapsed().as_secs_f64());
        return Ok(result);
    }

    let max_brute = 24;
    if crlf_positions.len() <= max_brute {
        eprintln!("Trying brute force...");
        if let Some(result) = try_brute_force(data, &crlf_positions, &lookup) {
            return Ok(result);
        }
    }

    bail!(
        "Recovery failed. {} CRLF positions, tried all strategies.",
        crlf_positions.len()
    )
}

fn validate_as_zip(data: &[u8]) -> bool {
    if !zip_util::is_valid_zip_signature(data) {
        return false;
    }
    match zip_util::parse_and_validate(data) {
        Ok(parsed) => parsed.entries.iter().all(|e| {
            e.crc32_actual.map_or(false, |actual| actual == e.crc32_expected)
        }) && !parsed.entries.is_empty(),
        Err(_) => false,
    }
}

struct CrlfLookup {
    index_at: Vec<Option<usize>>,
    positions: Vec<usize>,
}

impl CrlfLookup {
    fn new(crlf_positions: &[usize], data_len: usize) -> Self {
        let mut index_at = vec![None::<usize>; data_len];
        for (idx, &pos) in crlf_positions.iter().enumerate() {
            index_at[pos] = Some(idx);
        }
        CrlfLookup {
            index_at,
            positions: crlf_positions.to_vec(),
        }
    }
}

fn build_candidate_fast(data: &[u8], lookup: &CrlfLookup, remove_mask: &[bool]) -> Vec<u8> {
    let remove_count = remove_mask.iter().filter(|&&r| r).count();
    let mut result = Vec::with_capacity(data.len() - remove_count);
    for i in 0..data.len() {
        if let Some(idx) = lookup.index_at[i] {
            if remove_mask[idx] {
                continue;
            }
        }
        result.push(data[i]);
    }
    result
}

fn build_candidate(data: &[u8], crlf_positions: &[usize], remove_mask: &[bool]) -> Vec<u8> {
    let lookup = CrlfLookup::new(crlf_positions, data.len());
    build_candidate_fast(data, &lookup, remove_mask)
}

fn build_offset_map(corrupted: &[u8], crlf_positions: &[usize]) -> Vec<usize> {
    let mut map = Vec::with_capacity(corrupted.len());
    let mut ci = 0;
    for i in 0..corrupted.len() {
        if ci < crlf_positions.len() && crlf_positions[ci] == i {
            ci += 1;
        } else {
            map.push(i);
        }
    }
    map
}

fn try_decompress_check(data: &[u8], expected_crc: u32) -> bool {
    use flate2::read::DeflateDecoder;
    use std::io::Read;

    let mut decoder = DeflateDecoder::new(data);
    let mut result = Vec::new();
    if decoder.read_to_end(&mut result).is_err() {
        return false;
    }
    zip_util::crc32(&result) == expected_crc
}

fn try_remove_all(data: &[u8], crlf_positions: &[usize], lookup: &CrlfLookup) -> Option<RecoveryResult> {
    eprintln!("Strategy: remove all CR before LF");
    let remove_all: Vec<bool> = crlf_positions.iter().map(|_| true).collect();
    let candidate = build_candidate_fast(data, lookup, &remove_all);

    eprintln!(
        "  Candidate: {} bytes (corrupted: {}, removed {})",
        candidate.len(), data.len(), crlf_positions.len()
    );

    if validate_as_zip(&candidate) {
        return Some(RecoveryResult {
            data: candidate,
            strategy: "remove_all".to_string(),
            attempts: 1,
        });
    }
    None
}

fn try_per_file_fix(data: &[u8], crlf_positions: &[usize], lookup: &CrlfLookup) -> Option<RecoveryResult> {
    let remove_all_mask: Vec<bool> = crlf_positions.iter().map(|_| true).collect();
    let candidate = build_candidate_fast(data, lookup, &remove_all_mask);

    let parsed = match zip_util::parse_and_validate(&candidate) {
        Ok(p) => p,
        Err(e) => {
            eprintln!("  Cannot parse 'remove all': {}", e);
            return try_fix_global(data, crlf_positions, lookup);
        }
    };

    let bad_indices: Vec<usize> = parsed
        .entries
        .iter()
        .enumerate()
        .filter(|(_, e)| e.crc32_actual.map_or(true, |a| a != e.crc32_expected))
        .map(|(i, _)| i)
        .collect();

    if bad_indices.is_empty() {
        return Some(RecoveryResult {
            data: candidate,
            strategy: "remove_all".to_string(),
            attempts: 1,
        });
    }

    eprintln!(
        "  {} file(s) need fix: {:?}",
        bad_indices.len(),
        bad_indices.iter().map(|&i| parsed.entries[i].name.as_str()).collect::<Vec<_>>()
    );

    let offset_map = build_offset_map(data, crlf_positions);

    let mut file_comp_bounds: Vec<(usize, usize)> = Vec::new();
    let mut pos = 0;
    for entry in &parsed.entries {
        let header_size = 30 + entry.name.len();
        let comp_start = pos + header_size;
        let comp_end = comp_start + entry.compressed_size as usize;
        file_comp_bounds.push((comp_start, comp_end));
        pos = comp_end;
    }

    let mut kept_crlf_indices: Vec<usize> = Vec::new();
    let mut total_attempts = 0u64;

    for &file_idx in &bad_indices {
        let entry = &parsed.entries[file_idx];
        let (comp_start, comp_end) = file_comp_bounds[file_idx];
        let compressed_data = candidate[comp_start..comp_end].to_vec();

        eprintln!(
            "  '{}': {} bytes compressed, CRC {:08X}",
            entry.name, compressed_data.len(), entry.crc32_expected
        );

        let lf_positions: Vec<usize> = compressed_data
            .iter()
            .enumerate()
            .filter(|&(_, &b)| b == 0x0A)
            .map(|(i, _)| i)
            .collect();

        eprintln!("    {} LF positions in compressed data", lf_positions.len());

        let found = fix_section(
            &compressed_data,
            entry.crc32_expected,
            &lf_positions,
            &mut total_attempts,
        );

        match found {
            None => {
                eprintln!("    Could not fix '{}'", entry.name);
                return None;
            }
            Some(insert_positions) => {
                for rel_pos in &insert_positions {
                    let cand_pos = comp_start + rel_pos;
                    if cand_pos >= offset_map.len() {
                        eprintln!("    BUG: cand_pos {} >= offset_map.len() {}", cand_pos, offset_map.len());
                        return None;
                    }
                    let corr_pos = offset_map[cand_pos];
                    if corr_pos == 0 {
                        eprintln!("    BUG: corr_pos is 0 for cand_pos {}", cand_pos);
                        return None;
                    }
                    let crlf_corr_pos = corr_pos - 1;
                    if let Some(crlf_idx) = crlf_positions.iter().position(|&p| p == crlf_corr_pos) {
                        kept_crlf_indices.push(crlf_idx);
                    } else {
                        eprintln!(
                            "    WARNING: no CRLF at corrupted pos 0x{:X} (near candidate 0x{:X})",
                            crlf_corr_pos, cand_pos
                        );
                    }
                }
                eprintln!(
                    "    Fixed '{}': {} CRs restored",
                    entry.name, insert_positions.len()
                );
            }
        }
    }

    let mut final_mask: Vec<bool> = crlf_positions.iter().map(|_| true).collect();
    for &idx in &kept_crlf_indices {
        final_mask[idx] = false;
    }
    let candidate = build_candidate_fast(data, lookup, &final_mask);

    if !validate_as_zip(&candidate) {
        eprintln!("  Global candidate invalid, trying to patch remaining issues...");
        return try_patch_remaining(data, crlf_positions, lookup, &kept_crlf_indices);
    }

    Some(RecoveryResult {
        data: candidate,
        strategy: format!("per_file_fix({}_kept)", kept_crlf_indices.len()),
        attempts: total_attempts,
    })
}

fn try_patch_remaining(
    data: &[u8],
    crlf_positions: &[usize],
    lookup: &CrlfLookup,
    already_kept: &[usize],
) -> Option<RecoveryResult> {
    let kept_set: std::collections::HashSet<usize> = already_kept.iter().copied().collect();
    let remaining: Vec<usize> = (0..crlf_positions.len())
        .filter(|i| !kept_set.contains(i))
        .collect();

    eprintln!("  {} CRLFs not yet decided, trying to add more...", remaining.len());

    let mut mask: Vec<bool> = crlf_positions.iter().map(|_| true).collect();
    for &idx in already_kept {
        mask[idx] = false;
    }

    for &ri in &remaining {
        mask[ri] = false;
        let candidate = build_candidate_fast(data, lookup, &mask);
        if validate_as_zip(&candidate) {
            eprintln!("  Fixed by additionally keeping CR at index {}", ri);
            return Some(RecoveryResult {
                data: candidate,
                strategy: format!("patched({}_+1)", already_kept.len()),
                attempts: 1,
            });
        }
        mask[ri] = true;
    }

    if remaining.len() <= 300 {
        eprintln!("  Adding one more didn't work, trying pairs...");
        for i in 0..remaining.len() {
            mask[remaining[i]] = false;
            for j in (i + 1)..remaining.len() {
                mask[remaining[j]] = false;
                let candidate = build_candidate_fast(data, lookup, &mask);
                if validate_as_zip(&candidate) {
                    eprintln!("  Fixed by additionally keeping 2 CRs");
                    return Some(RecoveryResult {
                        data: candidate,
                        strategy: format!("patched({}_+2)", already_kept.len()),
                        attempts: 1,
                    });
                }
                mask[remaining[j]] = true;
            }
            mask[remaining[i]] = true;
        }
    }

    eprintln!("  Patching failed");
    None
}

fn fix_section(
    compressed_data: &[u8],
    expected_crc: u32,
    lf_positions: &[usize],
    total_attempts: &mut u64,
) -> Option<Vec<usize>> {
    if try_decompress_check(compressed_data, expected_crc) {
        return Some(vec![]);
    }

    // Keep one: insert 0x0D before one LF
    let t = Instant::now();
    for (k, &pos) in lf_positions.iter().enumerate() {
        *total_attempts += 1;
        let mut test = compressed_data.to_vec();
        test.insert(pos, 0x0D);
        if try_decompress_check(&test, expected_crc) {
            eprintln!(
                "    keep_one found at LF#{} (pos {}), {} attempts, {:.2}s",
                k, pos, *total_attempts, t.elapsed().as_secs_f64()
            );
            return Some(vec![pos]);
        }
    }
    eprintln!(
        "    keep_one failed: {} attempts, {:.2}s",
        lf_positions.len(),
        t.elapsed().as_secs_f64()
    );

    // Keep two: insert 0x0D before two LFs
    let t = Instant::now();
    let n = lf_positions.len();
    for i in 0..n {
        for j in (i + 1)..n {
            *total_attempts += 1;
            let mut test = compressed_data.to_vec();
            test.insert(lf_positions[j], 0x0D);
            test.insert(lf_positions[i], 0x0D);
            if try_decompress_check(&test, expected_crc) {
                eprintln!(
                    "    keep_two found at LF#{} + #{}, {} attempts, {:.2}s",
                    i, j, *total_attempts, t.elapsed().as_secs_f64()
                );
                return Some(vec![lf_positions[i], lf_positions[j]]);
            }
        }
    }
    eprintln!(
        "    keep_two failed: {} attempts, {:.2}s",
        n * (n - 1) / 2,
        t.elapsed().as_secs_f64()
    );

    // Keep three: insert 0x0D before three LFs
    let t = Instant::now();
    let limit = n.min(120);
    for i in 0..limit {
        for j in (i + 1)..limit {
            for k in (j + 1)..limit {
                *total_attempts += 1;
                let mut test = compressed_data.to_vec();
                test.insert(lf_positions[k], 0x0D);
                test.insert(lf_positions[j], 0x0D);
                test.insert(lf_positions[i], 0x0D);
                if try_decompress_check(&test, expected_crc) {
                    eprintln!(
                        "    keep_three found, {} attempts, {:.2}s",
                        *total_attempts, t.elapsed().as_secs_f64()
                    );
                    return Some(vec![lf_positions[i], lf_positions[j], lf_positions[k]]);
                }
            }
        }
    }
    eprintln!(
        "    keep_three failed (limit {}), {:.2}s",
        limit,
        t.elapsed().as_secs_f64()
    );

    None
}

fn try_fix_global(data: &[u8], crlf_positions: &[usize], lookup: &CrlfLookup) -> Option<RecoveryResult> {
    eprintln!("  Trying global keep-one...");
    let mut mask: Vec<bool> = crlf_positions.iter().map(|_| true).collect();
    for i in 0..crlf_positions.len() {
        mask[i] = false;
        let candidate = build_candidate_fast(data, lookup, &mask);
        if validate_as_zip(&candidate) {
            return Some(RecoveryResult {
                data: candidate,
                strategy: format!("keep_one_global_{}", i),
                attempts: (i + 1) as u64,
            });
        }
        mask[i] = true;
    }
    None
}

fn try_brute_force(data: &[u8], crlf_positions: &[usize], lookup: &CrlfLookup) -> Option<RecoveryResult> {
    let n = crlf_positions.len();
    let total = 1u64 << n;
    eprintln!("Brute force: {} positions, {} combos", n, total);

    for mask in 0..total {
        let remove: Vec<bool> = (0..n).map(|bit| (mask >> bit) & 1 == 0).collect();
        let candidate = build_candidate_fast(data, lookup, &remove);
        if validate_as_zip(&candidate) {
            let kept = remove.iter().filter(|&&r| !r).count();
            return Some(RecoveryResult {
                data: candidate,
                strategy: format!("brute_force(kept={})", kept),
                attempts: mask + 1,
            });
        }
    }
    None
}

use anyhow::{bail, Result};
use crc32fast::Hasher;
use flate2::write::DeflateEncoder;
use flate2::Compression;
use std::io::Write;

pub struct ZipEntry {
    pub name: String,
    pub data: Vec<u8>,
}

struct LocalHeaderInfo {
    offset: usize,
    compressed_size: u32,
    uncompressed_size: u32,
    crc32: u32,
    filename_len: u16,
    extra_len: u16,
    method: u16,
}

pub fn crc32(data: &[u8]) -> u32 {
    let mut hasher = Hasher::new();
    hasher.update(data);
    hasher.finalize()
}

fn deflate_compress(data: &[u8]) -> Vec<u8> {
    let mut encoder = DeflateEncoder::new(Vec::new(), Compression::default());
    encoder.write_all(data).unwrap();
    encoder.finish().unwrap()
}

pub fn create_zip(files: &[(&str, &[u8])]) -> Vec<u8> {
    let mut buf = Vec::new();
    let mut central_dir = Vec::new();
    let mut local_headers: Vec<LocalHeaderInfo> = Vec::new();

    for (name, data) in files {
        let name_bytes = name.as_bytes();
        let crc = crc32(data);
        let compressed = deflate_compress(data);
        let offset = buf.len() as u32;

        write_local_header(
            &mut buf,
            name_bytes,
            &compressed,
            crc,
            data.len() as u32,
        );

        local_headers.push(LocalHeaderInfo {
            offset: offset as usize,
            compressed_size: compressed.len() as u32,
            uncompressed_size: data.len() as u32,
            crc32: crc,
            filename_len: name_bytes.len() as u16,
            extra_len: 0,
            method: 8,
        });
    }

    let central_dir_offset = buf.len() as u32;

    for (i, (name, _)) in files.iter().enumerate() {
        let h = &local_headers[i];
        write_central_dir_entry(
            &mut central_dir,
            name.as_bytes(),
            h.offset as u32,
            h.compressed_size,
            h.uncompressed_size,
            h.crc32,
        );
    }

    buf.extend_from_slice(&central_dir);
    let central_dir_size = central_dir.len() as u32;

    write_eocd(
        &mut buf,
        files.len() as u16,
        central_dir_offset,
        central_dir_size,
    );

    buf
}

pub struct ParsedZip {
    pub entries: Vec<ParsedEntry>,
}

pub struct ParsedEntry {
    pub name: String,
    pub compressed_data: Vec<u8>,
    pub uncompressed_data: Option<Vec<u8>>,
    pub crc32_expected: u32,
    pub crc32_actual: Option<u32>,
    pub compressed_size: u32,
    pub uncompressed_size: u32,
    pub method: u16,
}

pub fn parse_and_validate(data: &[u8]) -> Result<ParsedZip> {
    let mut pos = 0;
    let mut entries = Vec::new();

    while pos + 4 <= data.len() {
        let sig = u32::from_le_bytes(data[pos..pos + 4].try_into().unwrap());
        if sig != 0x04034b50 {
            break;
        }
        if pos + 30 > data.len() {
            bail!("Local header truncated at offset {}", pos);
        }

        let method = u16::from_le_bytes(data[pos + 8..pos + 10].try_into().unwrap());
        let crc32_val = u32::from_le_bytes(data[pos + 14..pos + 18].try_into().unwrap());
        let comp_size = u32::from_le_bytes(data[pos + 18..pos + 22].try_into().unwrap());
        let uncomp_size = u32::from_le_bytes(data[pos + 22..pos + 26].try_into().unwrap());
        let name_len = u16::from_le_bytes(data[pos + 26..pos + 28].try_into().unwrap()) as usize;
        let extra_len = u16::from_le_bytes(data[pos + 28..pos + 30].try_into().unwrap()) as usize;

        let header_end = pos + 30 + name_len + extra_len;
        if header_end > data.len() {
            bail!("Header extends beyond data at offset {}", pos);
        }

        let name = String::from_utf8_lossy(&data[pos + 30..pos + 30 + name_len]).to_string();
        let data_start = header_end;
        let data_end = data_start + comp_size as usize;

        if data_end > data.len() {
            bail!(
                "Compressed data extends beyond file for '{}' (need {} bytes from offset {}, have {})",
                name, comp_size, data_start, data.len()
            );
        }

        let compressed_data = data[data_start..data_end].to_vec();

        let (uncompressed_data, crc32_actual) = if method == 8 {
            match try_decompress(&compressed_data) {
                Ok(decompressed) => {
                    let actual_crc = crc32(&decompressed);
                    (Some(decompressed), Some(actual_crc))
                }
                Err(_) => (None, None),
            }
        } else if method == 0 {
            let actual_crc = crc32(&compressed_data);
            (Some(compressed_data.clone()), Some(actual_crc))
        } else {
            (None, None)
        };

        entries.push(ParsedEntry {
            name,
            compressed_data,
            uncompressed_data,
            crc32_expected: crc32_val,
            crc32_actual,
            compressed_size: comp_size,
            uncompressed_size: uncomp_size,
            method,
        });

        pos = data_end;
    }

    Ok(ParsedZip { entries })
}

fn try_decompress(data: &[u8]) -> Result<Vec<u8>> {
    use flate2::read::DeflateDecoder;
    use std::io::Read;

    let mut decoder = DeflateDecoder::new(data);
    let mut result = Vec::new();
    decoder
        .read_to_end(&mut result)
        .map_err(|e| anyhow::anyhow!("Decompression failed: {}", e))?;
    Ok(result)
}

pub fn find_all_lf_positions(data: &[u8]) -> Vec<usize> {
    let mut positions = Vec::new();
    for i in 0..data.len() {
        if data[i] == 0x0A {
            positions.push(i);
        }
    }
    positions
}

pub fn find_crlf_positions(data: &[u8]) -> Vec<usize> {
    let mut positions = Vec::new();
    for i in 0..data.len().saturating_sub(1) {
        if data[i] == 0x0D && data[i + 1] == 0x0A {
            positions.push(i);
        }
    }
    positions
}

pub fn is_valid_zip_signature(data: &[u8]) -> bool {
    data.len() >= 4 && u32::from_le_bytes(data[0..4].try_into().unwrap()) == 0x04034b50
}
